import pickle
import networkx as nx

file_path = "./hybrid_db/knowledge_graph.pkl"

try:
    with open(file_path, 'rb') as f:
        G = pickle.load(f)
        
    if G is not None:
        # Everything that needs to be written to the file MUST be indented under this 'with'
        with open("./knowledge.txt", "w") as output_file:
            output_file.write(f"--- Graph Summary ---\n")
            output_file.write(f"Nodes: {G.number_of_nodes()}\n")
            output_file.write(f"Edges: {G.number_of_edges()}\n")
            
            output_file.write("\n--- Nodes (Entities) ---\n")
            for node, attributes in G.nodes(data=True):
                output_file.write(f"Node: {node} | Attributes: {attributes}\n")
            
            output_file.write("\n--- Edges (Relationships) ---\n")
            for source, target, attributes in G.edges(data=True):
                # Using .get() ensures it doesn't crash if 'relationship' key is missing
                rel = attributes.get('relationship', 'related to')
                output_file.write(f"{source} --[{rel}]--> {target}\n")
        
        print("Success! Data written to knowledge.txt")
    else:
        print("The loaded graph is None.")
except FileNotFoundError:
    print(f"Error: {file_path} not found.")
